------- TẠO TAB MENU (CHỨC NĂNG GAME) -------
local TabMenu = MakeTab({Name = "Menu"})

AddButton(TabMenu, {
  Name = "Bất Tử (God Mode)",
  Callback = function()
    -- Khóa trạng thái chết
    game.Players.LocalPlayer.Character.Humanoid:SetStateEnabled(Enum.HumanoidStateType.Dead, false)
    print("Đã bật Bất Tử")
  end
})

AddButton(TabMenu, {
  Name = "Bay Lên (Né Sóng)",
  Callback = function()
    local hrp = game.Players.LocalPlayer.Character.HumanoidRootPart
    hrp.CFrame = hrp.CFrame * CFrame.new(0, 50, 0) -- Bay lên 50 mét
  end
})

AddButton(TabMenu, {
  Name = "Hạ Xuống",
  Callback = function()
    local hrp = game.Players.LocalPlayer.Character.HumanoidRootPart
    hrp.CFrame = hrp.CFrame * CFrame.new(0, -50, 0) -- Hạ xuống 50 mét
  end
})

AddSlider(TabMenu, {
  Name = "Tốc Độ (Speed)",
  Min = 16,
  Max = 500,
  Default = 16,
  Callback = function(v)
    game.Players.LocalPlayer.Character.Humanoid.WalkSpeed = v
  end
})
